---------------------------- MODULE AbsJupiterH ----------------------------
EXTENDS AbsJupiter 
VARIABLE hlist
-------------------------------------------------------------
InitH == Init /\ hlist = {}
DoH(c) == Do(c) /\ hlist' = hlist \cup {list'[c]}
RevH(c) == Rev(c) /\ hlist' = hlist \cup {list'[c]}
SRevH == SRev /\ hlist' = hlist \cup {list'[Server]}
-------------------------------------------------------------
WLSpec == \A l1, l2 \in hlist: Compatible(l1, l2)
=============================================================================
\end{tla}
